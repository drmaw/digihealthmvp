import { db } from "/firebase.js"; import { collection, query, where, getDocs } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js"; export async function lookupByHealthId(healthId) { const q = query( collection(db, "users"), where("health_id_10", "==", healthId) ); const snap = await getDocs(q); if (snap.empty) return null; return snap.docs[0].data(); }
